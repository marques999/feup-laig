/**
 * @constructor
 * Creates an axis representing (X, Y, Z) referential. Colors are assigned in matching order (R, G, B), for easy interpretation
 * @param {WebGLRenderingContext} scene
 * @param {Number} length length of each axis. Default 1
 * @param {Number} thickness thickness of each axis. Default 0.2
 * 
 */
function CGFaxis(scene, length, thickness) {
	CGFobject.call(this,scene);

	//default case
	this.length = 5; 
	this.thickness = 0.05;
	
    switch (arguments.length ) {
        case 3: this.thickness = thickness;
        case 2: this.length = length;
    }

	this.HALF_PI = 3.1415926536/2;
	
	this.pyr = new CGFquadPyramid(scene,this.length-this.thickness/2.0,this.thickness);
	
};

CGFaxis.prototype = Object.create(CGFobject.prototype);
CGFaxis.prototype.constructor=CGFaxis;


/**
 * @method display
 * Displays the axis
 */
CGFaxis.prototype.display = function () {
	this.scene.pushMatrix();

	this.scene.setShininess(100.0);
	this.scene.setAmbient(1,1,1,1);

	//Z axiz - blue
	this.scene.pushMatrix();
	    this.scene.translate(0,0,this.thickness/2.0);
	    this.scene.setDiffuse(0,0,1,1);
	    this.scene.setSpecular(0,0,1,1);

	    this.pyr.display();
	this.scene.popMatrix();

    //X axiz - red
    this.scene.pushMatrix();
	    this.scene.rotate(this.HALF_PI,0,1,0);
	    this.scene.translate(0,0,this.thickness/2.0);
	    this.scene.setDiffuse(1,0,0,1);
        this.scene.setSpecular(1,0,0,1);
	    this.pyr.display();
	this.scene.popMatrix();

	//Y axiz - green
	this.scene.pushMatrix();
	    this.scene.rotate(-this.HALF_PI,1,0,0);
	    this.scene.translate(0,0,this.thickness/2.0);
	    this.scene.setDiffuse(0,1,0,1);
        this.scene.setSpecular(0,1,0,1);
	    this.pyr.display();
	this.scene.popMatrix();
	
	this.scene.popMatrix();
	
};



/**
 * @constructor
 * Creates a quadrangular pyramid. Used in CGFaxis.
 * @param {WebGLRenderingContext} scene
 * @param {Number} height height of the pyramid
 * @param {Number} side length of the side of the quadrangular pyramid
 * 
 */
function CGFquadPyramid(scene,height,side) {
	CGFobject.call(this,scene);
	
	this.halfSide = side/2;
	this.height = height;
	this.initBuffers();
	
};

CGFquadPyramid.prototype = Object.create(CGFobject.prototype);
CGFquadPyramid.prototype.constructor=CGFquadPyramid;

/**
 * @private
 * @ignore
 */
 
CGFquadPyramid.prototype.initBuffers = function () {
	this.vertices = [
            -this.halfSide, -this.halfSide, 0,
            this.halfSide, -this.halfSide, 0,
            -this.halfSide, this.halfSide, 0,
            this.halfSide, this.halfSide, 0,
            0, 0 , this.height
			];

	this.indices = [
            2, 1, 0,   //base
			1, 2, 3,   
			0, 1, 4,   //sides
			1, 3, 4,
			2, 0, 4,
			3, 2, 4
        ];
		
	this.primitiveType=this.scene.gl.TRIANGLES;
	this.initGLBuffers();
};
